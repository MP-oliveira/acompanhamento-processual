  match /databases/{database}/documents {
    // Usu치rios podem ler/escrever apenas seus pr칩prios dados
    match /users/{userId} {

    }
    
    // Processos - usu치rios autenticados podem ler/escrever
    match /processos/{processoId} {
      allow read, write: if request.auth != null;
    }
    
    // Alertas - usu치rios autenticados podem ler/escrever
    match /alertas/{alertaId} {
      allow read, write: if request.auth != null;
    }
  }
}
